% Makes \thetitle, \theauther available for the cover page.
\let\titleoriginal\title
\renewcommand{\title}[1]{
    \titleoriginal{#1}
    \newcommand{\thetitle}{#1}
}

\let\authororiginal\author
\renewcommand{\author}[1]{
    \authororiginal{#1}
    \newcommand{\theauthor}{#1}
}

\let\dateoriginal\date
\renewcommand{\date}[1]{
    \dateoriginal{#1}
    \newcommand{\thedate}{#1}
}

% Makes the \doublespacing macro available for the cover page.
\usepackage{setspace}

% Headers
\usepackage{fancyhdr}
\pagestyle{fancy}
\fancyhead{}
\fancyhead[R]{\thepage}
\fancyfoot{}
\renewcommand{\headrulewidth}{0pt}

% Place images exactly where they appear.
\usepackage{float}
\let\origfigure\figure
\let\endorigfigure\endfigure
\renewenvironment{figure}[1][2] {
    \expandafter\origfigure\expandafter[H]
} {
    \endorigfigure
}
